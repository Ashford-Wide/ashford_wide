{{- $src := .src -}} {{- $alt := .alt | default "" -}} {{- $class := .class | default "" -}} {{- $loading := .loading | default "lazy" -}} {{- $width := .width -}} {{- $quality := .quality -}} {{- $urlOnly := .urlOnly | default false -}} {{- $format := .format | default "avif" -}} {{/* Strip leading slash — resources.Get requires paths without it */}} {{- $assetPath := $src | strings.TrimPrefix "/" -}} {{- $img := resources.Get $assetPath -}} {{- if $img -}} {{- $qSuffix := "" -}} {{- with $quality }}{{- $qSuffix = printf " q%d" . -}}{{- end -}} {{- if $width -}} {{- $img = $img.Resize (printf "%dx %s%s" $width $format $qSuffix) -}} {{- else -}} {{- $img = $img.Process (printf "%s%s" $format $qSuffix) -}} {{- end -}} {{- if $urlOnly -}} {{- $img.RelPermalink -}} {{- else -}} {{- end -}} {{- else -}} {{/* Graceful fallback if file missing or resources.Get returns nil */}} {{- if $urlOnly -}} {{- $src -}} {{- else -}} {{- end -}} {{- end -}}
