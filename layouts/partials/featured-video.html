{{/* Renders a Vimeo facade for a given video ID. Params: videoID (required), videoTitle (optional) */}} {{- $videoID := .videoID -}} {{- with $videoID -}} {{- $videoTitle := $.videoTitle | default "Watch our video" -}} {{- $thumb := "" -}} {{- with resources.GetRemote (printf "https://vimeo.com/api/oembed.json?url=https://vimeo.com/%s" .) -}} {{- with . | transform.Unmarshal -}}{{- $thumb = .thumbnail_url -}}{{- end -}} {{- end -}}
{{- $facadeJS := resources.Get "js/vimeo-facade.js" -}} {{- if hugo.IsProduction -}}{{- $facadeJS = $facadeJS | minify | fingerprint -}}{{- end -}} {{- end -}}
